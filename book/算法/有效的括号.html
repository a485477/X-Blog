    # 20. 有效的括号

    给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。

    有效字符串需满足：

    1. 左括号必须用相同类型的右括号闭合。
    2. 左括号必须以正确的顺序闭合。

    注意空字符串可被认为是有效字符串。

    /** 
 * 解题思路
 * 还是利用栈结构先进先出的概念来解题
 * 遍历字符串，如果是左边符号推入到栈里面，如果是右边符号，则与栈最顶层判断是否对应，如果不对应则返回false
 * 最后如果栈里还有数据没处理完，说明传入的字符错误，返回false
 *
*/
let str = "{[]}"
var isValid = function(s) {
    let len = s.length
    let stack = []
    let map = {
        '(': ')',
        '[': ']',
        '{': '}'
    }

    for (let i in s) {
        let v = s[i]
        if (/\{|\[|\(/.test(v)) {
            stack.push(v)
        } else {
            let pop = stack.pop()
            if (map[pop] !== v) {
                return false
            }
        }
    }
    return stack.length === 0
};
console.log(isValid(str))

    # 优化思路

    /** 
 * 第一点 用变量缓存length，很关键，减少计算次数
 * 第二点 len % 2 判断是否是奇数， 不为0则为奇数，奇数肯定无法通过判定，就直接返回false，减少不必要的计算
 * !arr.length 直接可以取布尔值，比较巧妙
*/
var isValid = function(s) {
    let arr = []
    let len = s.length
    if (len%2) return false
    for (let i = 0; i < len; i++) {
        let letter = s[i]
        switch(letter) {
            case "(": {
                arr.push(letter)
                break;
            }
            case "[": {
                arr.push(letter)
                break;
            }
            case "{": {
                arr.push(letter)
                break;
            }
            case ")": {
                if (arr.pop() !== "(") return false
                break;
            }
            case "]": {
                 if (arr.pop() !== "[") return false
                break;
            }
            case "}": {
                if (arr.pop() !== "{") return false
                break;
            }
        }
    }
    return !arr.length

};
